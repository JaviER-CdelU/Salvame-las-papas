Estas reglas son una guía para INTEGRAR con tus reglas actuales; no reemplaces todo sin revisar.

Concepto necesario:

function signedIn() {
  return request.auth != null;
}

function ownsCommerce(comercioId) {
  return signedIn()
    && exists(/databases/$(database)/documents/comercios/$(comercioId))
    && get(/databases/$(database)/documents/comercios/$(comercioId)).data.ownerUid == request.auth.uid;
}

match /usuarios/{uid} {
  allow read, create, update: if signedIn() && request.auth.uid == uid;
}

match /pedidos/{pedidoId} {
  allow create: if signedIn()
    && request.resource.data.clienteUid == request.auth.uid;

  allow read: if signedIn()
    && (resource.data.clienteUid == request.auth.uid
        || ownsCommerce(resource.data.comercioId));

  allow update: if ownsCommerce(resource.data.comercioId);
  allow delete: if false;
}

La colección comercios debe conservar ownerUid para que ownsCommerce funcione.
El acceso administrador debe protegerse con un rol o Custom Claim; no alcanza con ocultar admin.html.
